  <h1>Silly Day Counter</h1>
  <p id="dayText">Click the button!</p>
  <button onclick="nextDay()">Add a Day</button>

  <!-- Confetti library -->
  <script src="https://cdn.jsdelivr.net/npm/canvas-confetti@1.9.3/dist/confetti.browser.min.js"></script>
  
  <script>
    let day = 0;
    
    function nextDay() {
      day++;
      if (day <= 7) {
        document.getElementById("dayText").textContent = day + " day" + (day > 1 ? "s" : "");
      }
      
      if (day === 7) {
        // Confetti blast
        confetti({
          particleCount: 300,
          spread: 120,
          origin: { y: 0.6 }
        });
        // Reset back to 0 after celebration
        setTimeout(() => {
          day = 0;
          document.getElementById("dayText").textContent = "Click the button!";
        }, 2000); // wait 2 seconds before resetting
      }
    }
  </script>
